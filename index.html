function openStockLoadModal() { let stockMap = {}; // accountStocksMap 객체를 순회하며 고유한 종목만 수집 (사용자/계좌 정보 제거) Object.values(accountStocksMap).forEach(stocks => { stocks.forEach(s => { // 종목명 기준으로 중복 제거 (이름이 같으면 덮어쓰기 형태로 하나만 유지) if (!stockMap[s.name]) { stockMap[s.name] = { name: s.name, code: s.code || "" }; } }); }); let listContainer = document.getElementById('stockLoadList'); listContainer.innerHTML = ''; let allStocks = Object.values(stockMap); if (allStocks.length === 0) { listContainer.innerHTML = '
등록된 종목이 없습니다.
'; } else { // 가나다순 정렬 후 렌더링 allStocks.sort((a, b) => a.name.localeCompare(b.name)).forEach(s => { let codeText = s.code ? `(${s.code})` : ''; let btn = document.createElement('button'); btn.style.cssText = "width:100%; text-align:left; padding:10px 15px; background:#f7fafc; border:1px solid #e2e8f0; border-radius:4px; cursor:pointer; font-size:14px; font-weight:bold; color:#4a5568; margin-bottom:4px; transition: background-color 0.2s;"; btn.innerHTML = `➕ ${s.name} ${codeText}`; btn.onmouseover = () => btn.style.backgroundColor = '#edf2f7'; btn.onmouseout = () => btn.style.backgroundColor = '#f7fafc'; btn.onclick = () => selectStockToLoad(s.name, s.code); listContainer.appendChild(btn); }); } document.getElementById('stockLoadModal').style.display = 'flex'; }
